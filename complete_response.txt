  // Load seller dashboard data
  async function loadSellerDashboard() {
    try {
      const sb = getSB();
      if (!sb) return;
      
      // Get current user
      const { data: { user } } = await sb.auth.getUser();
      if (!user) {
        console.log('No user logged in');
        return;
      }
      
      // Get seller's listings
      const { data: listings, error } = await sb
        .from('auction_items')
        .select('*, bids(count)')
        .eq('seller_id', user.id)
        .order('created_at', { ascending: false });
      
      if (error) throw error;
      
      // Calculate stats
      const activeListings = listings.filter(l => l.status === 'active').length;
      const soldListings = listings.filter(l => l.status === 'sold').length;
      const totalEarned = listings
        .filter(l => l.status === 'sold' && l.current_bid)
        .reduce((sum, l) => sum + (l.current_bid || 0), 0);
      const avgPrice = activeListings > 0 
        ? listings
            .filter(l => l.status === 'active' && l.current_bid)
            .reduce((sum, l) => sum + (l.current_bid || 0), 0) / activeListings
        : 0;
      
      // Update stats UI
      const statsEl = document.getElementById('seller-stats');
      if (statsEl) {
        statsEl.innerHTML = `
          <div class="stat-card">
            <div class="stat-value">${activeListings}</div>
            <div class="stat-label">Active Listings</div>
          </div>
          <div class="stat-card">
            <div class="stat-value">${soldListings}</div>
            <div class="stat-label">Sold Items</div>
          </div>
          <div class="stat-card">
            <div class="stat-value">$${totalEarned.toFixed(2)}</div>
            <div class="stat-label">Total Earned</div>
          </div>
          <div class="stat-card">
            <div class="stat-value">$${avgPrice.toFixed(2)}</div>
            <div class="stat-label">Avg. Price</div>
          </div>
        `;
      }
      
      // Update active listings
      const activeListEl = document.getElementById('active-listings');
      if (activeListEl) {
        const activeItems = listings.filter(l => l.status === 'active');
        if (activeItems.length > 0) {
          activeListEl.innerHTML = activeItems.map(item => `
            <div class="listing-row">
              <div class="listing-thumb" style="background:linear-gradient(135deg,#f0e8d8,#e0d0b8)">
                ${item.category === 'Furniture' ? '🪑' : 
                  item.category === 'Art' ? '🖼️' :
                  item.category === 'Jewelry' ? '💍' :
                  item.category === 'Lighting' ? '🪔' :
                  item.category === 'Rugs' ? '🏺' : '🧸'}
              </div>
              <div class="listing-content">
                <div class="listing-name">${item.title}</div>
                <div class="listing-meta">
                  <span class="listing-cat">${item.category}</span>
                  <span class="listing-time">${formatTimeAgo(item.created_at)}</span>
                </div>
                <div class="listing-stats">
                  <span class="listing-views">👁️ ${item.view_count || 0} views</span>
                  <span class="listing-bids">🤝 ${item.bid_count || 0} bids</span>
                </div>
              </div>
              <div class="listing-bid">$${item.current_bid || item.starting_bid || 0}</div>
            </div>
          `).join('');
        } else {
          activeListEl.innerHTML = '<div class="empty-state">No active listings</div>';
        }
      }
      
      // Update recent activity
      const activityEl = document.getElementById('recent-activity');
      if (activityEl && listings.length > 0) {
        const recentItems = listings.slice(0, 5);
        activityEl.innerHTML = recentItems.map(item => `
          <div class="act-row">
            <div class="act-icon">${item.status === 'active' ? '📈' : '💰'}</div>
            <div class="act-content">
              <div class="act-title">${item.title}</div>
              <div class="act-desc">${item.status === 'active' ? 'New bid received' : 'Item sold'} • ${formatTimeAgo(item.updated_at)}</div>
            </div>
            <div class="act-amt">$${item.current_bid || item.starting_bid || 0}</div>
          </div>
        `).join('');
      }
      
    } catch (e) {
      console.error('loadSellerDashboard:', e);
    }
  }
  
  // Helper function to format time ago
  function formatTimeAgo(dateString) {
    const date = new Date(dateString);
    const now = new Date();
    const diffMs = now - date;
    const diffMins = Math.floor(diffMs / 60000);
    const diffHours = Math.floor(diffMs / 3600000);
    const diffDays = Math.floor(diffMs / 86400000);
    
    if (diffMins < 60) return `${diffMins}m ago`;
    if (diffHours < 24) return `${diffHours}h ago`;
    if (diffDays < 7) return `${diffDays}d ago`;
    return date.toLocaleDateString();
  }
EOF

cat > /tmp/admin_function.js << 'EOF'
  // Load admin panel data
  async function loadAdminPanel() {
    try {
      const sb = getSB();
      if (!sb) return;
      
      // Get admin stats - all data
      const [
        { data: users, error: usersError },
        { data: items, error: itemsError },
        { data: bids, error: bidsError },
        { data: revenueData, error: revenueError }
      ] = await Promise.all([
        sb.from('profiles').select('id, created_at'),
        sb.from('auction_items').select('id, status, current_bid, created_at'),
        sb.from('bids').select('id, amount, created_at'),
        sb.from('auction_items').select('current_bid').eq('status', 'sold')
      ]);
      
      if (usersError) throw usersError;
      if (itemsError) throw itemsError;
      if (bidsError) throw bidsError;
      if (revenueError) throw revenueError;
      
      // Calculate stats
      const totalUsers = users.length;
      const activeItems = items.filter(i => i.status === 'active').length;
      const totalBids = bids.length;
      const totalRevenue = revenueData.reduce((sum, item) => sum + (item.current_bid || 0), 0);
      
      // Get recent users (last 7 days)
      const weekAgo = new Date();
      weekAgo.setDate(weekAgo.getDate() - 7);
      const newUsers = users.filter(u => new Date(u.created_at) > weekAgo).length;
      
      // Update stats UI
      const statsEl = document.getElementById('admin-stats');
      if (statsEl) {
        statsEl.innerHTML = `
          <div class="stat-card">
            <div class="stat-value">${totalUsers}</div>
            <div class="stat-label">Total Users</div>
            <div class="stat-change">+${newUsers} this week</div>
          </div>
          <div class="stat-card">
            <div class="stat-value">${activeItems}</div>
            <div class="stat-label">Active Items</div>
            <div class="stat-change">${items.length} total listed</div>
          </div>
          <div class="stat-card">
            <div class="stat-value">${totalBids}</div>
            <div class="stat-label">Total Bids</div>
            <div class="stat-change">${bids.length} placed</div>
          </div>
          <div class="stat-card">
            <div class="stat-value">$${totalRevenue.toFixed(2)}</div>
            <div class="stat-label">Total Revenue</div>
            <div class="stat-change">From ${revenueData.length} sold items</div>
          </div>
        `;
      }
      
      // Update recent activity
      const activityEl = document.getElementById('admin-activity');
      if (activityEl) {
        // Get recent bids with user info
        const { data: recentBids, error: bidsError2 } = await sb
          .from('bids')
          .select(`
            id,
            amount,
            created_at,
            auction_items(title),
            profiles(full_name)
          `)
          .order('created_at', { ascending: false })
          .limit(10);
        
        if (!bidsError2 && recentBids) {
          activityEl.innerHTML = recentBids.map(bid => `
            <div class="act-row">
              <div class="act-icon">💰</div>
              <div class="act-content">
                <div class="act-title">New bid on "${bid.auction_items?.title || 'Item'}"</div>
                <div class="act-desc">By ${bid.profiles?.full_name || 'User'} • ${formatTimeAgo(bid.created_at)}</div>
              </div>
              <div class="act-amt">$${bid.amount}</div>
            </div>
          `).join('');
        }
      }
      
      // Update user management table
      const usersTable = document.getElementById('users-table');
      if (usersTable) {
        // Get users with their activity
        const { data: userList, error: userListError } = await sb
          .from('profiles')
          .select(`
            id,
            full_name,
            email,
            created_at,
            auction_items(count),
            bids(count)
          `)
          .order('created_at', { ascending: false })
          .limit(20);
        
        if (!userListError && userList) {
          usersTable.innerHTML = userList.map(user => `
            <tr>
              <td>
                <div class="user-cell">
                  <div class="user-av">${(user.full_name || 'U').charAt(0).toUpperCase()}</div>
                  <div class="user-info">
                    <div class="user-name">${user.full_name || 'User'}</div>
                    <div class="user-email">${user.email || 'No email'}</div>
                  </div>
                </div>
              </td>
              <td>${new Date(user.created_at).toLocaleDateString()}</td>
              <td>${user.auction_items?.[0]?.count || 0}</td>
              <td>${user.bids?.[0]?.count || 0}</td>
              <td>
                <select class="role-select" onchange="updateUserRole('${user.id}', this.value)">
                  <option value="user">User</option>
                  <option value="seller">Seller</option>
                  <option value="admin">Admin</option>
                </select>
              </td>
            </tr>
          `).join('');
        }
      }
      
    } catch (e) {
      console.error('loadAdminPanel:', e);
    }
  }
  
  // Placeholder for updateUserRole function
  async function updateUserRole(userId, role) {
    console.log(`Updating user ${userId} to role ${role}`);
    // In a real app, you would update the user's role in the database
  }
EOF

// Helper function to format time ago
function formatTimeAgo(dateString) {
  if (!dateString) return 'Just now';
  const date = new Date(dateString);
  const now = new Date();
  const diffMs = now - date;
  const diffMins = Math.floor(diffMs / 60000);
  const diffHours = Math.floor(diffMs / 3600000);
  const diffDays = Math.floor(diffMs / 86400000);
  
  if (diffMins < 60) return `${diffMins}m ago`;
  if (diffHours < 24) return `${diffHours}h ago`;
  if (diffDays < 7) return `${diffDays}d ago`;
  return date.toLocaleDateString();
}

// Helper function to format time ago\
function formatTimeAgo(dateString) {\
  if (!dateString) return "Just now";\
  const date = new Date(dateString);\
  const now = new Date();\
  const diffMs = now - date;\
  const diffMins = Math.floor(diffMs / 60000);\
  const diffHours = Math.floor(diffMs / 3600000);\
  const diffDays = Math.floor(diffMs / 86400000);\
  \
  if (diffMins < 60) return `${diffMins}m ago`;\
  if (diffHours < 24) return `${diffHours}h ago`;\
  if (diffDays < 7) return `${diffDays}d ago`;\
  return date.toLocaleDateString();\
}\

// Helper function to format time ago
function formatTimeAgo(dateString) {
  if (!dateString) return 'Just now';
  const date = new Date(dateString);
  const now = new Date();
  const diffMs = now - date;
  const diffMins = Math.floor(diffMs / 60000);
  const diffHours = Math.floor(diffMs / 3600000);
  const diffDays = Math.floor(diffMs / 86400000);
  
  if (diffMins < 60) return `${diffMins}m ago`;
  if (diffHours < 24) return `${diffHours}h ago`;
  if (diffDays < 7) return `${diffDays}d ago`;
  return date.toLocaleDateString();
}

// Helper function to format time ago
function formatTimeAgo(dateString) {
  if (!dateString) return 'Just now';
  const date = new Date(dateString);
  const now = new Date();
  const diffMs = now - date;
  const diffMins = Math.floor(diffMs / 60000);
  const diffHours = Math.floor(diffMs / 3600000);
  const diffDays = Math.floor(diffMs / 86400000);
  
  if (diffMins < 60) return `${diffMins}m ago`;
  if (diffHours < 24) return `${diffHours}h ago`;
  if (diffDays < 7) return `${diffDays}d ago`;
  return date.toLocaleDateString();
}
